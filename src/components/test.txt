mutation CreatePerson(
  $name:String!
  $gender:GenderType
  $biography:String
  $birthday:DateTime
  $adult:Boolean
  $placeOfBirth:String
  $alsoKnownAs:[String]
){
  createPerson(
    data:{
      name:$name
      gender:$gender
      biography:$biography
      birthday:$birthday
      adult:$adult
      placeOfBirth:$placeOfBirth
      alsoKnownAs:$alsoKnownAs
    }){
    data {
      id
      name
      gender
      biography
      birthday
      adult
      placeOfBirth
      alsoKnownAs
    }
  }
}
